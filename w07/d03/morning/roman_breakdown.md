function romanize(inputNum){
  var romanize = '';
  if(inputNum >= 9) romanize += inputNum
  if(inputNum >= 5) return "V";
  if(inputNum >= 4) return "IV";
  while (inputNum > 0){
    romanice += "I"
    inputNum -= 1
  }
  return romanize;
}
